## On Frontend

If you have a web app with multiple pages where each one triggers a new page refresh then you will most likely need the basic JavaScript method. This means you will not need to create separate actions which trigger when the Mycroft will appear or update: The page refresh itself will trigger the action.

To get the Mycroft to appear on your web app copy and paste the snippet below before the `</body>` tag or in the `<head>` tag on every page where you want the Mycroft to appear for website visitors or in your Node.js file.

In the code, you will need your own appID (as shared by the Mycroft team) at the placeholder that says `Your-App-ID`.


<AccordionGroup>
  <Accordion title="Code for open-access site" icon="globe">
    ```javascript init.js
<script>
  const APP_ID = "Your-App-ID" // IMPORTANT: Replace with your App ID

  window.mycroftSettings = {
    api_base: "https://codenet.mycroft.tech",
    app_id: APP_ID,
  };
</script>

<script>
        (function () {
            const w = window, d = document, mc = w.Mycroft;

            if (typeof mc === "function") {
                mc('update', w.mycroftSettings);  // Update settings if already initialized
            } else {
                w.Mycroft = function () { w.Mycroft.q.push(arguments); };  // Queue for API calls
                w.Mycroft.q = [];  // Initialize call queue

                const loadScript = function () {
                    fetch(w.mycroftSettings.api_base + '?appID=' + w.mycroftSettings.app_id)
                        .then(res => res.ok ? res.text() : Promise.reject('Script load failed: ' + res.statusText))
                        .then(code => { let s = d.createElement('script'); s.text = code; d.head.appendChild(s); })
                        .catch(err => console.error(err));
                };

                if (d.readyState === "complete") loadScript();
                else w.addEventListener ? w.addEventListener('load', loadScript) : w.attachEvent('onload', loadScript);
            }
        })();(function(){var w=window;var ic=w.Intercom;if(typeof ic==="function"){ic('update',w.intercomSettings);}else{var d=document;var i=function(){i.c(arguments);};i.q=[];i.c=function(args){i.q.push(args);};w.Intercom=i;var l=function(){var s=d.createElement('script');s.type='text/javascript';s.async=true;s.src='https://widget.intercom.io/widget/ ' + APP_ID;var x=d.getElementsByTagName('script')[0];x.parentNode.insertBefore(s, x);};if(document.readyState==='complete'){l();}else if(w.attachEvent){w.attachEvent('onload',l);}else{w.addEventListener('load',l,false);}}})();


</script>
```
